    <script lang="ts">
      import { Button, Modal, PinInput } from "@dxdns/feflow"
      import { onMount } from "svelte"

      function handleOpen() {
        isOpen = !isOpen
      }

      function handleClose() {
        isOpen = false
      }

        isOpen = true
      })
    </script>

    <Modal {isOpen} {handleClose}>

        <h3>Two-Factor Verification</h3>
      </Modal.Header>

        </p>

        <div
          style="
                display: flex; 
                align-items: center; 
                justify-content: center;
                overflow: hidden;
                flex-wrap: nowrap;
                gap: 0.5rem;
                "
        >
          {#each Array.from(Array(4)) as _, i}
            <PinInput isLoading={i === 0} />
          {/each}
        </div>
      </Modal.Content>
      <Modal.Actions class="border-0" align="center">
        <Button
          class="hover-border-on-error hover-text-on-error"
          variant="outlined"
          onclick={handleClose}
        >
          Cancel
        </Button>
        <Button onclick={handleClose}>Save</Button>
      </Modal.Actions>
    </Modal>
